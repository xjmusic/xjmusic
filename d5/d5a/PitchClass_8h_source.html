69  static PitchClass pitchClassOf(const std::string& name) {
70  if (name.empty())

72  std::string normalized = accidentalNormalized(name);
73  std::string atBeginning = StringUtils::match(pitchClassAtBeginning, normalized).value_or("");
74  if (pitchClassOfString.find(atBeginning) != pitchClassOfString.end())
75  return pitchClassOfString[atBeginning];
